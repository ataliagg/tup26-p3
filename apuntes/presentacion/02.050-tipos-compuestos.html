Tipos Compuestos en C

Arrays, tuplas y colecciones

Arrays

Un array es una colección de elementos del mismo tipo con acceso por índice.

// Declaración e inicialización
int[] numeros = { 1, 2, 3, 4, 5 };
string[] nombres = new string[3];  // 3 elementos, inicialmente null

// Acceso por índice (comienza en 0)
Console.WriteLine(numeros[0]);    // 1
numeros[2] = 99;

// Propiedades y métodos
numeros.Length;                    // cantidad de elementos
numeros.GetLength(0);             // longitud en primera dimensión

Arrays multidimensionales

// Matriz 2D
int[,] matriz = {
    { 1, 2, 3 },
    { 4, 5, 6 },
    { 7, 8, 9 }
};

Console.WriteLine(matriz[1, 2]);  // 6
Console.WriteLine(matriz.GetLength(0));  // 3 filas
Console.WriteLine(matriz.GetLength(1));  // 3 columnas

// Arrays de arrays (jagged)
int[][] jagged = new int[3][];
jagged[0] = new int[5];
jagged[1] = new int[3];

Tuplas

Una tupla agrupa varios valores de posiblemente tipos distintos.

// Tupla con nombres
var persona = (Nombre: "Ana", Edad: 25, Ciudad: "Rosario");
Console.WriteLine(persona.Nombre);   // "Ana"
Console.WriteLine(persona.Edad);     // 25

// Tupla sin nombres
var punto = ("Coordenada X", 10, 20);
Console.WriteLine(punto.Item1);      // "Coordenada X"
Console.WriteLine(punto.Item2);      // 10

// Desestructuración
var (nombre, edad, ciudad) = persona;
Console.WriteLine(nombre);  // "Ana"

(string a, int b) tupla = ("hola", 42);

Tuplas en métodos

// Retornar múltiples valores
static (int Suma, int Producto) Calcular(int a, int b) {
    return (a + b, a * b);
}

var resultado = Calcular(5, 3);
Console.WriteLine(resultado.Suma);      // 8
Console.WriteLine(resultado.Producto);  // 15

// Con desestructuración
var (suma, producto) = Calcular(5, 3);

List

Colección dinámica con métodos útiles.

var numeros = new List<int> { 1, 2, 3 };
var nombres = new List<string>();

// Acceso y modificación
numeros[0];           // acceso por índice
numeros.Add(4);       // agregar elemento
numeros.Insert(1, 10); // insertar en posición
numeros.Remove(2);    // quitar elemento

// Propiedades
numeros.Count;        // cantidad de elementos
numeros.Capacity;     // capacidad interna

// Búsqueda
bool contiene = numeros.Contains(3);
int posicion = numeros.IndexOf(3);

// Limpieza
numeros.Clear();

Dictionary<K,V>

Colección de pares clave-valor con búsqueda rápida.

var edades = new Dictionary<string, int> {
    ["Ana"] = 25,
    ["Carlos"] = 30,
    ["Laura"] = 28
};

// Acceso
Console.WriteLine(edades["Ana"]);       // 25
edades["Ana"] = 26;                     // modificar

// Agregar
edades.Add("Pedro", 35);

// Búsqueda segura
if (edades.TryGetValue("Ana", out int edad))
    Console.WriteLine($"Ana tiene {edad}");

// Iteración
foreach (var (nombre, edad) in edades)
    Console.WriteLine($"{nombre}: {edad}");

HashSet

Colección de valores únicos sin orden específico.

var colores = new HashSet<string> { "rojo", "azul", "verde" };

colores.Add("rojo");        // no se agrega (ya existe)
colores.Add("amarillo");    // ✓ nuevo

colores.Contains("azul");   // true
colores.Remove("rojo");     // true

// Operaciones de conjunto
var otros = new HashSet<string> { "azul", "violeta" };
colores.IntersectWith(otros);  // solo los que están en ambos
colores.UnionWith(otros);      // todos los únicos

SortedSet y SortedDictionary<K,V>

Colecciones ordenadas.

// SortedSet — sin duplicados, siempre ordenado
var numeros = new SortedSet<int> { 3, 1, 4, 1, 5 };
// { 1, 3, 4, 5 }

// SortedDictionary — pares ordenados por clave
var resultados = new SortedDictionary<string, int> {
    ["Carlos"] = 85,
    ["Ana"] = 92,
    ["Laura"] = 78
};

// La iteración respeta el orden
foreach (var (nombre, puntaje) in resultados)
    Console.WriteLine($"{nombre}: {puntaje}");

Métodos útiles comunes

var numeros = new List<int> { 3, 1, 4, 1, 5, 9 };

// Búsqueda
numeros.Find(n => n > 5);        // primer elemento > 5
numeros.FindAll(n => n > 3);     // todos los > 3
numeros.Exists(n => n == 4);     // ¿existe 4?

// Ordenamiento
numeros.Sort();                  // ordena in-place
numeros.Sort((a, b) => b.CompareTo(a));  // descendente

// Transformación
var dobles = numeros.ConvertAll(n => n * 2);

// Recorrido
numeros.ForEach(n => Console.WriteLine(n));

// Verificación
bool todosPares = numeros.All(n => n % 2 == 0);
bool algunoPositivo = numeros.Any(n => n > 0);

Structs vs Classes

// Struct — tipo por valor
public struct Punto {
    public int X { get; set; }
    public int Y { get; set; }
}

// Class — tipo por referencia
public class Persona {
    public string Nombre { get; set; }
    public int Edad { get; set; }
}

Punto p1 = new Punto { X = 1, Y = 2 };
Punto p2 = p1;
p2.X = 10;
// p1.X sigue siendo 1 — struct es por valor

Persona per1 = new Persona { Nombre = "Ana" };
Persona per2 = per1;
per2.Nombre = "Carlos";
// per1.Nombre es "Carlos" — class es por referencia

Records

Una forma moderna de definir tipos de datos simples.

// Record — inmutable por defecto
record Producto(string Nombre, decimal Precio, int Stock);

var p1 = new Producto("Monitor", 450_000, 5);
Console.WriteLine(p1.Nombre);

// Desestructuración
var (nombre, precio, stock) = p1;

// Non-destructive update
var p2 = p1 with { Stock = 10 };
